\begin{frame}
	\frametitle{Plantilla desde \TeX{}studio}
	\begin{enumerate}
		\item

		      Abra el programa \TeX{}studio y seleccione la pestaña
		      \alert{Archivo}.

		\item

		      Seleccione \alert{Nuevo desde Plantilla} y en la nueva
		      ventana desde el panel lateral escoja \alert{Libro}.

		\item

		      Guarde el fichero y reemplaza los campos de \alert{Nombre}
		      y \alert{Título} por su nombre y el título de un libro que
		      haya escrito.
	\end{enumerate}

	\begin{figure}[ht!]
		\centering
		\includegraphics[width=.75\paperwidth]{texstudio}
	\end{figure}
\end{frame}

\begin{frame}[fragile]
	\setbeamercovered{invisible}
	\frametitle{Estilo \mintinline{latex}|standalone|}

	\begin{enumerate}
		\item

		      Guarde un archivo en blanco con el nombre
		      \verb|helloworld.tex|.

		\item

		      Digite en el fichero el contenido que se muestra debajo y luego
		      comente la primera línea
		      \mintinline{latex}|\documentclass{standalone}| y descomente
		      la segunda línea \mintinline{latex}|\documentclass{minimal}|
		      ¿Cuál es la diferencia entre estos?
	\end{enumerate}

	\begin{columns}
		\begin{column}{.58\paperwidth}
			\inputminted[fontsize=\tiny,linenos,frame=single,escapeinside=||]{latex}{helloworld} % tex-fmt: skip
		\end{column}
		\pause
		\begin{column}{.38\paperwidth}
			\begin{figure}[ht!]
				\centering
				\includegraphics[width=.25\paperwidth]{helloworld}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}
